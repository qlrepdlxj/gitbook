library(plotly)

data = read.csv("/home/ubuntu/movie/data/85579.csv")
data = data.frame(data)
# data
x1 <- data$집계날짜
y1 <- data$일일관람객
y2 <- data$검색량
y3 <- data$리뷰수
y4 <- data$평점/10

ay <- list(
  tickfont = list(color = "red"),
  overlaying = "y",
  side = "right",
  title = ""
)
p <- plot_ly(x = ~1:length(x1), y = y1, name = "일일관람객",type='bar') %>%
  add_lines(x = ~1:length(x1), y = y2, name = "검색량", yaxis = "y2") %>%
  layout(
    title = "", yaxis2 = ay,
    xaxis = list(title="")
  )
